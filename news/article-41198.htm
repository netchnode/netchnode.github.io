使用vue-server-renderer编译，插入html到模版中

const fs=require('fs')const{     resolve}=require('path')const createBundleRenderer=require('vue-server-renderer').createBundleRenderer// 读取`skeleton.json`，以`index.html`为模板写入内容const renderer=createBundleRenderer(resolve(__dirname,'./dist/skeleton.json'),{     template: fs.readFileSync(resolve(__dirname,'./index.html'),'utf-8')})// 把上一步模板完成的内容写入（替换）`index.html` renderer.renderToString({},(err, html)=>{if(err){         console.log('err:'+ err);}     fs.writeFileSync('index.html', html,'utf-8')})

原来是原来html中有这样的代码

<%= htmlWebpackPlugin.options.loading.css%><%= htmlWebpackPlugin.options.loading.html%>

导致解析插入失败

<!--vue-ssr-outlet-->

替换这个，把<%= htmlWebpackPlugin.options.loading.css %> <%= htmlWebpackPlugin.options.loading.html %>
删除掉就好了。
